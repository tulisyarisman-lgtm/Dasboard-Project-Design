Konek ke spreadsheet biar tim bisa edit barengan.

Dapat dari deploy Apps Script kamu (lihat panduan di bawah).

📖 Panduan setup (klik untuk buka)
  1. Buka sheets.new — bikin Google Sheet kosong, kasih nama (mis. "Cimory Project Sync").
  2. Di Sheet itu, klik menu Extensions → Apps Script.
  3. Hapus kode default yang muncul, lalu copy seluruh kode di bawah dan tempel di editor Apps Script:
const SHEET_KEY = 'cimory_data';
function doGet(e){
  const action = (e.parameter.action||'load');
  if(action==='load') return load_();
  if(action==='ping') return json_({ok:true,version:'1.0'});
  return json_({error:'Unknown action'});
}
function doPost(e){
  let body={};
  try{ body = JSON.parse(e.postData.contents); }catch(err){ return json_({error:'Bad JSON'}); }
  const action = body.action||'';
  if(action==='save') return save_(body);
  if(action==='ping') return json_({ok:true,version:'1.0'});
  return json_({error:'Unknown action'});
}
function sheet_(){
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName('Data');
  if(!sh){
    sh = ss.insertSheet('Data');
    sh.getRange('A1:B1').setValues([['key','value']]);
    sh.setFrozenRows(1);
  }
  return sh;
}
function readCell_(key){
  const sh = sheet_();
  const data = sh.getRange(2,1,Math.max(1,sh.getLastRow()),2).getValues();
  for(const [k,v] of data){ if(k===key) return v; }
  return '';
}
function writeCell_(key,value){
  const sh = sheet_();
  const last = Math.max(1, sh.getLastRow());
  const data = sh.getRange(2,1,last,1).getValues();
  for(let i=0;i<data.length;i++){
    if(data[i][0]===key){ sh.getRange(i+2,2).setValue(value); return; }
  }
  sh.appendRow([key,value]);
}
function load_(){
  const projects     = JSON.parse(readCell_('projects')||'[]');
  const logs         = JSON.parse(readCell_('logs')||'[]');
  const coachingForms= JSON.parse(readCell_('coachingForms')||'{}');
  const lastUpdate   = readCell_('lastUpdate')||'';
  return json_({projects, logs, coachingForms, lastUpdate});
}
function save_(body){
  const now = new Date().toISOString();
  writeCell_('projects',     JSON.stringify(body.projects||[]));
  writeCell_('logs',         JSON.stringify(body.logs||[]));
  writeCell_('coachingForms',JSON.stringify(body.coachingForms||{}));
  writeCell_('lastUpdate',   now);
  return json_({ok:true, lastUpdate:now});
}
function json_(obj){
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
  1. Save (Ctrl+S) → kasih nama project bebas.
  2. Klik Deploy → New deployment (atau ikon gear → Web app).
  3. Setting deployment:
    • Type: Web app
    • Execute as: Me (akun kamu)
    • Who has access: Anyone ← penting!
  4. Klik Deploy → Google minta authorize → izinkan.
  5. Copy Web app URL (biasanya: https://script.google.com/macros/s/AKfycb…/exec).
  6. Paste URL itu ke kolom di atas → klik Konek & Sync.

⚠️ URL ini = akses ke data kamu. Bagi cuma ke tim. Kalau bocor, redeploy untuk dapat URL baru.
